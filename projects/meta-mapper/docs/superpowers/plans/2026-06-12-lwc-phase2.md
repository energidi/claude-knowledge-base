> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

- [ ] **Step 2: Create metaMapperTree.js**

```javascript
import { LightningElement, api, track } from 'lwc';
import { applyFilters, buildNodeMap } from 'c/metaMapperNodeUtils';

const ROW_HEIGHT = 36;
const OVERSCAN = 5;
const INDENT_PER_LEVEL = 20;
const SEARCH_STORAGE_KEY = 'metaMapper_treeSearch_v1';

const TYPE_ICONS = {
    ApexClass:      'utility:apex',
    ApexTrigger:    'utility:apex',
    Flow:           'utility:flow',
    CustomField:    'utility:custom_apps',
    ValidationRule: 'utility:rules',
    WorkflowRule:   'utility:process',
    Report:         'utility:report'
};

const TYPE_COLORS = {
    ApexClass:      '#0176d3',
    ApexTrigger:    '#0176d3',
    Flow:           '#1b5297',
    CustomField:    '#2e844a',
    ValidationRule: '#ba0517',
    WorkflowRule:   '#b35a00', // corrected per CONTRAST_MATRIX.md
    Report:         '#444444'
};

export default class MetaMapperTree extends LightningElement {
    @api filters = {};
    @api selectedNodeId = null;

    @track searchTerm = '';
    @track expandedIds = new Set();
    @track visibleRows = [];
    @track showContextMenu = false;
    @track contextMenuX = 0;
    @track contextMenuY = 0;
    @track maxLevelFilter = 9999;
    @track typeFilters = []; // active type filters (empty = all)

    _allNodes = [];
    _nodeMap = new Map();
    _flatRows = []; // all renderable rows (after filter + expand/collapse)
    _scrollTop = 0;
    _containerHeight = 400;
    _activeIndex = -1;
    _contextNodeId = null;
    _hasRendered = false;
    _lastFocusBeforeMenu = null;

    @api
    set nodes(val) {
        this._allNodes = val || [];
        this._nodeMap = buildNodeMap(this._allNodes);
        // Load search state from sessionStorage
        try {
            const stored = sessionStorage.getItem(SEARCH_STORAGE_KEY);
            if (stored) this.searchTerm = stored;
        } catch (_e) { /* ignore */ }
        this._rebuild();
    }

    get nodes() { return this._allNodes; }

    connectedCallback() {
        window.addEventListener('click', this._handleOutsideClick.bind(this));
        window.addEventListener('resize', this._updateContainerHeight.bind(this));
    }

    disconnectedCallback() {
        window.removeEventListener('click', this._handleOutsideClick.bind(this));
        window.removeEventListener('resize', this._updateContainerHeight.bind(this));
    }

    renderedCallback() {
        this._updateContainerHeight();
        if (!this._hasRendered && this._flatRows.length > 0) {
            this._hasRendered = true;
            this.dispatchEvent(new CustomEvent('tabready'));
        }
        // Attach scroll listener once
        const container = this.refs?.treeContainer;
        if (container && !container._scrollBound) {
            container._scrollBound = true;
            container.addEventListener('scroll', () => {
                this._scrollTop = container.scrollTop;
                this._updateVisibleRows();
            });
        }
    }

    // --- Core build ---

    _rebuild() {
        const filtered = this._applyAllFilters();
        this._flatRows = this._buildFlatRows(filtered, null, 0);
        this._updateVisibleRows();
    }

    _applyAllFilters() {
        let nodes = applyFilters(this._allNodes, this.filters || {});
        // Additional local search filter
        if (this.searchTerm.trim()) {
            const term = this.searchTerm.trim().toLowerCase();
            nodes = nodes.filter(n => (n.Metadata_Name__c || '').toLowerCase().includes(term));
        }
        return nodes;
    }

    _buildFlatRows(nodes, parentId, depth) {
        const rows = [];
        const children = nodes.filter(n =>
            parentId === null ? !n.Parent_Dependency__c : n.Parent_Dependency__c === parentId
        );
        children.forEach(node => {
            const hasChildren = nodes.some(n => n.Parent_Dependency__c === node.Id);
            const expanded = this.expandedIds.has(node.Id) || depth === 0;
            rows.push(this._buildRow(node, depth, hasChildren, expanded));
            if (expanded && hasChildren) {
                rows.push(...this._buildFlatRows(nodes, node.Id, depth + 1));
            }
        });
        return rows;
    }

    _buildRow(node, depth, hasChildren, expanded) {
        const isSelected = node.Metadata_Id__c === this.selectedNodeId;
        const type = node.Metadata_Type__c || 'default';
        const color = TYPE_COLORS[type] || '#3e3e3c';
        const showLowConfidence = node.Discovery_Source__c === 'Supplemental'
            && node.Supplemental_Confidence__c != null
            && node.Supplemental_Confidence__c < 70;
        const ariaExpanded = !hasChildren ? undefined : expanded ? 'true' : 'false';
        return {
            id: node.Id,
            Metadata_Id__c: node.Metadata_Id__c,
            Metadata_Name__c: node.Metadata_Name__c,
            depth,
            ariaLevel: depth + 1,
            ariaExpanded,
            hasChildren,
            expanded,
            icon: TYPE_ICONS[type] || 'utility:connected_apps',
            iconStyle: `color:${color}`,
            indentStyle: `padding-left:${depth * INDENT_PER_LEVEL + 8}px`,
            rowClass: `tree-row${isSelected ? ' is-selected' : ''}${node.Is_Circular__c ? ' is-circular' : ''}`,
            displayLabel: node.Metadata_Name__c || node.Metadata_Id__c,
            expandIcon: expanded ? 'utility:chevrondown' : 'utility:chevronright',
            toggleLabel: expanded ? `Collapse ${node.Metadata_Name__c}` : `Expand ${node.Metadata_Name__c}`,
            Is_Circular__c: node.Is_Circular__c,
            Is_Dynamic_Reference__c: node.Is_Dynamic_Reference__c,
            isSupplemental: node.Discovery_Source__c === 'Supplemental',
            showLowConfidence,
            confidenceTitle: `Low confidence match (${node.Supplemental_Confidence__c}%) - verify manually`,
            confidenceAriaLabel: `Warning: low confidence supplemental match, ${node.Supplemental_Confidence__c}% confidence`
        };
    }

    _updateVisibleRows() {
        const totalRows = this._flatRows.length;
        const startIndex = Math.max(0, Math.floor(this._scrollTop / ROW_HEIGHT) - OVERSCAN);
        const endIndex = Math.min(totalRows, startIndex + Math.ceil(this._containerHeight / ROW_HEIGHT) + OVERSCAN * 2);
        this.visibleRows = this._flatRows.slice(startIndex, endIndex);
    }

    _updateContainerHeight() {
        const el = this.refs?.treeContainer;
        if (el) this._containerHeight = el.clientHeight || 400;
    }

    // --- Getters ---

    get topSpacerStyle() {
        const startIndex = Math.max(0, Math.floor(this._scrollTop / ROW_HEIGHT) - OVERSCAN);
        return `height:${startIndex * ROW_HEIGHT}px`;
    }

    get bottomSpacerStyle() {
        const totalRows = this._flatRows.length;
        const startIndex = Math.max(0, Math.floor(this._scrollTop / ROW_HEIGHT) - OVERSCAN);
        const endIndex = Math.min(totalRows, startIndex + Math.ceil(this._containerHeight / ROW_HEIGHT) + OVERSCAN * 2);
        return `height:${(totalRows - endIndex) * ROW_HEIGHT}px`;
    }

    get isZeroVisible() { return this._flatRows.length === 0 && this._allNodes.length > 0; }
    get maxDepthAvailable() {
        return this._allNodes.reduce((acc, n) => Math.max(acc, n.Dependency_Depth__c || 0), 0);
    }

    get collapseSubtreeLabel() {
        if (this._contextNodeId) {
            const row = this._flatRows.find(r => r.Metadata_Id__c === this._contextNodeId);
            if (row && row.depth === 0) return 'Collapse all children';
        }
        return 'Collapse subtree';
    }

    get typeFilterOptions() {
        const types = [...new Set(this._allNodes.map(n => n.Metadata_Type__c).filter(Boolean))].sort();
        return types.map(t => ({
            type: t,
            label: t,
            active: this.typeFilters.includes(t),
            count: this._allNodes.filter(n => n.Metadata_Type__c === t).length,
            pillClass: `type-pill${this.typeFilters.includes(t) ? ' is-active' : ''}`
        }));
    }

    get contextMenuStyle() { return `left:${this.contextMenuX}px;top:${this.contextMenuY}px`; }

    // --- Event handlers ---

    handleSearchInput(event) {
        this.searchTerm = event.target.value;
        try { sessionStorage.setItem(SEARCH_STORAGE_KEY, this.searchTerm); } catch (_e) { /* ignore */ }
        this._rebuild();
    }

    clearSearch() {
        this.searchTerm = '';
        try { sessionStorage.removeItem(SEARCH_STORAGE_KEY); } catch (_e) { /* ignore */ }
        this._rebuild();
    }

    handleTypeFilterToggle(event) {
        const type = event.currentTarget.dataset.type;
        if (this.typeFilters.includes(type)) {
            this.typeFilters = this.typeFilters.filter(t => t !== type);
        } else {
            this.typeFilters = [...this.typeFilters, type];
        }
        this._rebuild();
    }

    handleLevelFilter(event) {
        this.maxLevelFilter = parseInt(event.target.value, 10);
        this._rebuild();
    }

    handleRowClick(event) {
        const rowEl = event.currentTarget;
        const nodeId = rowEl.dataset.id;
        const row = this._flatRows.find(r => r.id === nodeId);
        if (!row) return;
        this.dispatchEvent(new CustomEvent('nodeselected', {
            detail: { nodeId: row.Metadata_Id__c }, bubbles: true, composed: true
        }));
    }

    handleRowFocus(event) {
        const nodeId = event.currentTarget.dataset.id;
        const idx = this._flatRows.findIndex(r => r.id === nodeId);
        if (idx >= 0) this._activeIndex = idx;
    }

    handleToggleExpand(event) {
        event.stopPropagation();
        const nodeId = event.currentTarget.dataset.id;
        if (this.expandedIds.has(nodeId)) {
            this.expandedIds.delete(nodeId);
        } else {
            this.expandedIds.add(nodeId);
        }
        this.expandedIds = new Set(this.expandedIds); // trigger reactivity
        this._rebuild();
    }

    handleKeyDown(event) {
        const rows = this._flatRows;
        if (!rows.length) return;
        if (event.key === 'ArrowDown') {
            event.preventDefault();
            this._activeIndex = Math.min(rows.length - 1, this._activeIndex + 1);
            this._focusRow(this._activeIndex);
        } else if (event.key === 'ArrowUp') {
            event.preventDefault();
            this._activeIndex = Math.max(0, this._activeIndex - 1);
            this._focusRow(this._activeIndex);
        } else if (event.key === 'Enter' || event.key === ' ') {
            event.preventDefault();
            const row = rows[this._activeIndex];
            if (row) {
                this.dispatchEvent(new CustomEvent('nodeselected', {
                    detail: { nodeId: row.Metadata_Id__c }, bubbles: true, composed: true
                }));
            }
        } else if (event.key === 'ArrowRight') {
            const row = rows[this._activeIndex];
            if (row && row.hasChildren && !row.expanded) {
                this.expandedIds.add(row.id);
                this.expandedIds = new Set(this.expandedIds);
                this._rebuild();
            }
        } else if (event.key === 'ArrowLeft') {
            const row = rows[this._activeIndex];
            if (row && row.hasChildren && row.expanded) {
                this.expandedIds.delete(row.id);
                this.expandedIds = new Set(this.expandedIds);
                this._rebuild();
            }
        }
    }

    _focusRow(index) {
        const rows = this.template.querySelectorAll('[role="treeitem"]');
        const visibleStart = Math.max(0, Math.floor(this._scrollTop / ROW_HEIGHT) - OVERSCAN);
        const localIndex = index - visibleStart;
        if (localIndex >= 0 && localIndex < rows.length) {
            rows[localIndex].focus();
        }
    }

    handleContextMenu(event) {
        event.preventDefault();
        const rowEl = event.target.closest('[role="treeitem"]');
        if (!rowEl) return;
        const nodeId = rowEl.dataset.id;
        const row = this._flatRows.find(r => r.id === nodeId);
        if (!row) return;
        this._lastFocusBeforeMenu = document.activeElement;
        this._contextNodeId = row.Metadata_Id__c;
        this.contextMenuX = event.clientX;
        this.contextMenuY = event.clientY;
        this.showContextMenu = true;
        setTimeout(() => {
            const menu = this.refs?.contextMenu;
            if (menu) {
                const firstItem = menu.querySelector('[role="menuitem"]');
                if (firstItem) firstItem.focus();
            }
        }, 0);
    }

    handleMenuKeyDown(event) {
        if (event.key === 'Escape') {
            this.showContextMenu = false;
            if (this._lastFocusBeforeMenu) this._lastFocusBeforeMenu.focus();
        }
    }

    _handleOutsideClick(event) {
        const menu = this.refs?.contextMenu;
        if (menu && !menu.contains(event.target)) {
            this.showContextMenu = false;
        }
    }

    handleCopyApiName() {
        this.showContextMenu = false;
        const node = this._nodeMap.get(this._contextNodeId);
        if (!node) return;
        navigator.clipboard.writeText(node.Metadata_Name__c || '').then(() => {
            this.dispatchEvent(new CustomEvent('showerror', {
                detail: { message: `Copied API Name: ${node.Metadata_Name__c}`, variant: 'success' },
                bubbles: true, composed: true
            }));
        }).catch(() => {
            this.dispatchEvent(new CustomEvent('showerror', {
                detail: { message: 'Could not copy to clipboard. Select and copy the name manually instead.' },
                bubbles: true, composed: true
            }));
        });
    }

    handleCollapseSubtree() {
        this.showContextMenu = false;
        const row = this._flatRows.find(r => r.Metadata_Id__c === this._contextNodeId);
        if (!row) return;
        if (row.depth === 0) {
            // Collapse all — remove all expanded IDs except root
            this.expandedIds = new Set();
        } else {
            this._collapseSubtree(row.id);
        }
        this._rebuild();
    }

    _collapseSubtree(nodeId) {
        this.expandedIds.delete(nodeId);
        const children = this._flatRows.filter(r => {
            // Find direct children
            const node = this._allNodes.find(n => n.Id === r.id);
            return node && node.Parent_Dependency__c === nodeId;
        });
        children.forEach(c => this._collapseSubtree(c.id));
    }

    handleViewPathInGraph() {
        this.showContextMenu = false;
        this.dispatchEvent(new CustomEvent('viewpathtograph', {
            detail: { nodeId: this._contextNodeId }, bubbles: true, composed: true
        }));
    }
}
```

```javascript
import { LightningElement, api, track, wire } from 'lwc';
import { loadScript } from 'lightning/platformResourceLoader';
import ECharts from '@salesforce/resourceUrl/ECharts';
import { applyFilters, buildNodeMap } from 'c/metaMapperNodeUtils';
import { renderPills } from 'c/metaMapperUtils';

const NODE_COLORS = {
    ApexClass:      '#0176d3',
    ApexTrigger:    '#0176d3',
    Flow:           '#1b5297',
    CustomField:    '#2e844a',
    ValidationRule: '#ba0517',
    WorkflowRule:   '#b35a00', // corrected per CONTRAST_MATRIX.md
    Report:         '#444444',
    default:        '#3e3e3c'
};
const NODE_ICONS = {
    ApexClass:      'utility:apex',
    ApexTrigger:    'utility:apex',
    Flow:           'utility:flow',
    CustomField:    'utility:custom_apps',
    ValidationRule: 'utility:rules',
    WorkflowRule:   'utility:process',
    Report:         'utility:report',
    default:        'utility:connected_apps'
};
const LARGE_GRAPH_THRESHOLD = 8000;
const EXPAND_GUARD_THRESHOLD = 1000;

export default class MetaMapperGraph extends LightningElement {
    @api filters = {};
    @api selectedNodeId = null;

    @track chartLoading = true;
    @track graphSearch = '';
    @track showContextMenu = false;
    @track contextMenuX = 0;
    @track contextMenuY = 0;
    @track showExpandModal = false;
    @track showShortcuts = false;
    @track focusPathActive = false;
    @track showSpanningNotice = true;
    @track largeGraphDismissed = false;
    @track ariaTableBusy = 'false';
    @track ariaTableRows = [];
    @track isMobile = window.innerWidth < 1024;

    _allNodes = [];
    _nodeMap = new Map();
    _chart = null;
    _echartsLoaded = false;
    _contextNodeId = null;
    _activeNodeIndex = -1;
    _focusedNodeIds = new Set();
    _tabReadyFired = false;
    _ariaRebuildTimer = null;
    _lastFocusBeforeMenu = null;
    _lastFocusBeforeShortcuts = null;

    @api
    set nodes(val) {
        this._allNodes = val || [];
        this._nodeMap = buildNodeMap(this._allNodes);
        if (this._echartsLoaded) this._renderChart();
        this._scheduleAriaRebuild();
    }
    get nodes() { return this._allNodes; }

    connectedCallback() {
        loadScript(this, ECharts + '/echarts.min.js')
            .then(() => {
                this._echartsLoaded = true;
                this._registerTheme();
                this._renderChart();
            })
            .catch(err => {
                console.error('ECharts load error:', err);
                this.chartLoading = false;
            });
        window.addEventListener('click', this._handleOutsideClick.bind(this));
        window.addEventListener('resize', this._handleResize.bind(this));
        // Spanning notice: check localStorage
        const dismissed = localStorage.getItem('metaMapper_spanningTreeNotice_v1');
        if (dismissed) this.showSpanningNotice = false;
    }

    disconnectedCallback() {
        if (this._chart) { this._chart.dispose(); this._chart = null; }
        window.removeEventListener('click', this._handleOutsideClick.bind(this));
        window.removeEventListener('resize', this._handleResize.bind(this));
        clearTimeout(this._ariaRebuildTimer);
    }

    // --- ECharts ---

    _registerTheme() {
        /* global echarts */
        echarts.registerTheme('sfDark', {
            backgroundColor: '#1B1B1B',
            textStyle: { color: '#FFFFFF' },
            title: { textStyle: { color: '#FFFFFF' } },
            legend: { textStyle: { color: '#FFFFFF' } }
        });
    }

    _renderChart() {
        const container = this.refs?.echartsContainer;
        if (!container || !this._echartsLoaded) return;

        const isDark = document.body.classList.contains('slds-theme_inverse');
        if (!this._chart) {
            this._chart = echarts.init(container, isDark ? 'sfDark' : null);
            this._chart.on('click', (params) => {
                if (params.dataType === 'node') {
                    this._onNodeClick(params.data.id);
                }
            });
            this._chart.on('contextmenu', (params) => {
                if (params.dataType === 'node') {
                    params.event.event.preventDefault();
                    this._openContextMenu(params.data.id, params.event.event.clientX, params.event.event.clientY);
                }
            });
        }

        const filteredNodes = applyFilters(this._allNodes, this.filters || {});
        const { nodes, links } = this._buildGraphData(filteredNodes);

        const option = {
            series: [{
                type: 'graph',
                layout: 'force',
                animation: true,
                data: nodes,
                links,
                roam: true,
                label: { show: true, position: 'right', formatter: '{b}' },
                force: { repulsion: 200, edgeLength: 80 },
                lineStyle: { color: '#c9c9c9' },
                emphasis: { itemStyle: { borderColor: '#FFB81C', borderWidth: 3, shadowBlur: 8 } }
            }]
        };

        this._chart.setOption(option, true);
        this._chart.on('finished', () => {
            this.chartLoading = false;
            if (!this._tabReadyFired) {
                this._tabReadyFired = true;
                this.dispatchEvent(new CustomEvent('tabready'));
            }
        });
    }

    _buildGraphData(nodes) {
        const gNodes = nodes.map(n => ({
            id: n.Id,
            name: n.Metadata_Name__c || n.Metadata_Id__c,
            value: n.Metadata_Name__c,
            Metadata_Id__c: n.Metadata_Id__c,
            category: n.Metadata_Type__c,
            symbolSize: 20,
            itemStyle: {
                color: NODE_COLORS[n.Metadata_Type__c] || NODE_COLORS.default,
                borderType: n.Is_Circular__c ? 'dashed' : 'solid',
                borderWidth: n.Is_Circular__c ? 2 : 1,
                borderColor: n.Is_Circular__c ? '#706e6b' : undefined
            },
            tooltip: {
                formatter: `${n.Metadata_Name__c} (${n.Metadata_Type__c})` +
                    (n.Dependency_Context__c ? ` | ${renderPills(n.Dependency_Context__c)}` : '') +
                    (n.Supplemental_Confidence__c != null ? ` | Confidence: ${n.Supplemental_Confidence__c}% - verify manually` : '')
            }
        }));

        const nodeIds = new Set(nodes.map(n => n.Id));
        const gLinks = nodes
            .filter(n => n.Parent_Dependency__c && nodeIds.has(n.Parent_Dependency__c))
            .map(n => ({ source: n.Parent_Dependency__c, target: n.Id }));

        return { nodes: gNodes, links: gLinks };
    }

    _onNodeClick(nodeId) {
        const node = this._allNodes.find(n => n.Id === nodeId);
        if (!node) return;
        this.dispatchEvent(new CustomEvent('nodeselected', {
            detail: { nodeId: node.Metadata_Id__c }, bubbles: true, composed: true
        }));
    }

    _openContextMenu(nodeId, x, y) {
        this._lastFocusBeforeMenu = document.activeElement;
        if (this.focusPathActive) { this.clearFocusPath(); }
        this._contextNodeId = nodeId;
        this.contextMenuX = x;
        this.contextMenuY = y;
        this.showContextMenu = true;
        setTimeout(() => {
            const menu = this.refs?.contextMenu;
            const firstItem = menu && menu.querySelector('[role="menuitem"]');
            if (firstItem) firstItem.focus();
        }, 0);
    }

    // --- Computed getters ---

    get nodeCount() { return this._allNodes.length; }
    get showLargeGraphWarning() { return !this.largeGraphDismissed && this.nodeCount > LARGE_GRAPH_THRESHOLD && !this.isMobile; }
    get expandAllLabel() { return `Expand All (${this.nodeCount} nodes)`; }
    get expandAllAriaLabel() { return `Confirm. Expand all ${this.nodeCount} nodes now. This may slow your browser.`; }
    get contextMenuStyle() { return `left:${this.contextMenuX}px;top:${this.contextMenuY}px`; }
    get graphAriaLabel() { return `Dependency graph for ${this._allNodes[0]?.Metadata_Name__c || 'this component'}. Use Tree View tab for full keyboard access and screen reader support.`; }

    get legendItems() {
        return Object.entries(NODE_COLORS)
            .filter(([k]) => k !== 'default')
            .map(([type, color]) => ({
                type, label: type,
                icon: NODE_ICONS[type] || 'utility:connected_apps',
                swatchStyle: `background:${color};width:12px;height:12px;display:inline-block;border-radius:2px`
            }));
    }

    // --- Keyboard navigation (virtual focus index) ---

    handleCanvasKeyDown(event) {
        const wrapper = this.refs?.canvasWrapper;
        if (!wrapper || !wrapper.contains(document.activeElement)) return;

        if (event.key === 'Control' && event.key === 'k') {
            // handled by Ctrl+K below
        }
        if ((event.ctrlKey || event.metaKey) && event.key === 'k') {
            event.stopPropagation();
            event.preventDefault();
            const searchInput = this.refs?.graphSearchInput;
            if (searchInput) { searchInput.focus(); searchInput.select(); }
            return;
        }
        if (event.shiftKey && event.key === '?') {
            event.preventDefault();
            this._lastFocusBeforeShortcuts = document.activeElement;
            this.showShortcuts = true;
            return;
        }
        if (event.key === 'Escape') {
            if (this.graphSearch) { this.graphSearch = ''; this._applyGraphSearch(); return; }
            if (this.focusPathActive) { this.clearFocusPath(); return; }
        }

        const nodes = applyFilters(this._allNodes, this.filters || {});
        if (!nodes.length) return;

        if (event.key === 'ArrowDown' || event.key === 'ArrowRight') {
            event.preventDefault();
            this._activeNodeIndex = Math.min(nodes.length - 1, this._activeNodeIndex + 1);
            this._highlightVirtualFocus(nodes[this._activeNodeIndex]);
        } else if (event.key === 'ArrowUp' || event.key === 'ArrowLeft') {
            event.preventDefault();
            this._activeNodeIndex = Math.max(0, this._activeNodeIndex - 1);
            this._highlightVirtualFocus(nodes[this._activeNodeIndex]);
        } else if (event.key === 'Enter') {
            event.preventDefault();
            const node = nodes[this._activeNodeIndex];
            if (node) this._onNodeClick(node.Id);
        }
    }

    _highlightVirtualFocus(node) {
        if (!node || !this._chart) return;
        this._chart.dispatchAction({
            type: 'highlight',
            dataIndex: applyFilters(this._allNodes, this.filters || {}).findIndex(n => n.Id === node.Id)
        });
        const region = this.refs?.graphLiveRegion;
        if (region) region.textContent = `Focused: ${node.Metadata_Name__c}`;
    }

    // --- Search ---

    handleSearchKeyDown(event) {
        if (event.key === 'Escape') { this.graphSearch = ''; this._applyGraphSearch(); }
    }

    handleGraphSearch(event) {
        this.graphSearch = event.target.value;
        this._applyGraphSearch();
    }

    _applyGraphSearch() {
        if (!this._chart) return;
        const term = this.graphSearch.trim().toLowerCase();
        const nodes = applyFilters(this._allNodes, this.filters || {});
        if (!term) {
            this._chart.dispatchAction({ type: 'downplay' });
            const region = this.refs?.graphLiveRegion;
            if (region) region.textContent = '';
            return;
        }
        const matchCount = nodes.filter(n => (n.Metadata_Name__c || '').toLowerCase().includes(term)).length;
        // Highlight matching nodes via emphasis
        nodes.forEach((n, i) => {
            const matches = (n.Metadata_Name__c || '').toLowerCase().includes(term);
            this._chart.dispatchAction({ type: matches ? 'highlight' : 'downplay', dataIndex: i });
        });
        const region = this.refs?.graphLiveRegion;
        if (region) region.textContent = `${matchCount} nodes match your search`;
    }

    // --- Focus path ---

    handleFocusPath() {
        this.showContextMenu = false;
        if (!this._contextNodeId || !this._chart) return;
        const node = this._allNodes.find(n => n.Metadata_Id__c === this._contextNodeId);
        if (!node) return;
        // Build ancestor chain from Ancestor_Path__c
        const ancestors = node.Ancestor_Path__c ? node.Ancestor_Path__c.split('|') : [];
        this._focusedNodeIds = new Set([...ancestors, this._contextNodeId]);
        this.focusPathActive = true;
        // Dim non-focused nodes
        const nodes = applyFilters(this._allNodes, this.filters || {});
        nodes.forEach((n, i) => {
            const focused = this._focusedNodeIds.has(n.Metadata_Id__c);
            this._chart.dispatchAction({ type: focused ? 'highlight' : 'downplay', dataIndex: i });
        });
        const region = this.refs?.graphLiveRegion;
        if (region) region.textContent = `Focus path activated: ${this._focusedNodeIds.size} nodes highlighted`;
        // Move keyboard focus to Clear Focus button
        setTimeout(() => {
            const btn = this.template.querySelector('[aria-label="Clear focus path"]');
            if (btn) btn.focus();
        }, 0);
    }

    clearFocusPath() {
        this.focusPathActive = false;
        this._focusedNodeIds = new Set();
        if (this._chart) this._chart.dispatchAction({ type: 'downplay' });
    }

    // --- Context menu actions ---

    handleCopyApiName() {
        this.showContextMenu = false;
        const node = this._allNodes.find(n => n.Metadata_Id__c === this._contextNodeId);
        if (!node) return;
        navigator.clipboard.writeText(node.Metadata_Name__c || '').then(() => {
            this.dispatchEvent(new CustomEvent('showerror', {
                detail: { message: `Copied API Name: ${node.Metadata_Name__c}`, variant: 'success' },
                bubbles: true, composed: true
            }));
        }).catch(() => {
            this.dispatchEvent(new CustomEvent('showerror', {
                detail: { message: 'Could not copy to clipboard. Select and copy the name manually instead.' },
                bubbles: true, composed: true
            }));
        });
    }

    handleCollapseSubtreeMenu() {
        this.showContextMenu = false;
        // Graph collapse = visual only via ECharts collapse action
        if (this._chart) {
            // ECharts force graph doesn't support collapse natively — update option to remove descendants
            const node = this._allNodes.find(n => n.Metadata_Id__c === this._contextNodeId);
            if (!node) return;
            this._renderChartWithCollapse(node.Id);
        }
    }

    _renderChartWithCollapse(rootNodeId) {
        const collapsedIds = this._getSubtreeIds(rootNodeId);
        const filtered = applyFilters(this._allNodes, this.filters || {}).filter(n => !collapsedIds.has(n.Id));
        const { nodes, links } = this._buildGraphData(filtered);
```

```javascript
import { LightningElement, api, track } from 'lwc';
import { isNamespacePrefixed } from 'c/metaMapperNodeUtils';
import { buildDefaultFilename } from 'c/metaMapperUtils';

// Metadata type → package.xml type name mapping
const PACKAGE_TYPE_MAP = {
    ApexClass:      'ApexClass',
    ApexTrigger:    'ApexTrigger',
    Flow:           'Flow',
    CustomField:    'CustomField',
    ValidationRule: 'ValidationRule',
    WorkflowRule:   'WorkflowRule',
    Report:         'Report'
};

export default class MetaMapperExport extends LightningElement {
    @api allNodes = [];
    @api nodes = []; // filtered nodes
    @api job = null;
    @track showAdvanced = false;

    get exportDisabled() { return !this.nodes || this.nodes.length === 0; }
    get advancedToggleIcon() { return this.showAdvanced ? 'utility:chevronup' : 'utility:chevrondown'; }
    get targetApiName() { return (this.job && this.job.Target_API_Name__c) || 'scan'; }

    toggleAdvanced() { this.showAdvanced = !this.showAdvanced; }

    handleDownloadCsv() {
        try {
            const rows = this._buildCsvRows(this.nodes);
            const csv = rows.map(r => r.map(cell => `"${String(cell || '').replace(/"/g, '""')}"`).join(',')).join('\n');
            this._download(this._filename('csv'), 'text/csv;charset=utf-8;', csv);
        } catch (err) {
            this._showError('Export failed. Try filtering to fewer nodes first, or use JSON instead of CSV for large result sets.');
        }
    }

    _buildCsvRows(nodes) {
        const header = ['Level', 'Metadata_Type', 'Metadata_Name', 'Metadata_ID', 'Parent_Name', 'Is_Circular', 'Is_Dynamic'];
        const idToName = new Map(nodes.map(n => [n.Id, n.Metadata_Name__c]));
        const rows = nodes.map(n => [
            n.Dependency_Depth__c,
            n.Metadata_Type__c,
            n.Metadata_Name__c,
            n.Metadata_Id__c,
            n.Parent_Dependency__c ? (idToName.get(n.Parent_Dependency__c) || n.Parent_Dependency__c) : '',
            n.Is_Circular__c ? 'true' : 'false',
            n.Is_Dynamic_Reference__c ? 'true' : 'false'
        ]);
        return [header, ...rows];
    }

    handleDownloadJson() {
        try {
            const json = JSON.stringify(this._buildJsonTree(this.nodes), null, 2);
            this._download(this._filename('json'), 'application/json', json);
        } catch (err) {
            this._showError('Export failed. Try filtering to fewer nodes first, or use JSON instead of CSV for large result sets.');
        }
    }

    _buildJsonTree(nodes) {
        const idMap = new Map(nodes.map(n => [n.Id, { ...n, children: [] }]));
        const roots = [];
        idMap.forEach(n => {
            if (n.Parent_Dependency__c && idMap.has(n.Parent_Dependency__c)) {
                idMap.get(n.Parent_Dependency__c).children.push(n);
            } else {
                roots.push(n);
            }
        });
        return roots;
    }

    handleDownloadPackageXml() {
        try {
            const xml = this._buildPackageXml(this.nodes);
            this._download(this._filename('xml', 'package'), 'application/xml', xml);
        } catch (err) {
            this._showError('Export failed. Try filtering to fewer nodes first, or use JSON instead of CSV for large result sets.');
        }
    }

    _buildPackageXml(nodes) {
        // Group by type, exclude namespace-prefixed
        const typeMap = {};
        nodes.forEach(n => {
            if (!PACKAGE_TYPE_MAP[n.Metadata_Type__c]) return;
            if (isNamespacePrefixed(n.Metadata_Name__c, n.Metadata_Type__c)) return;
            const t = PACKAGE_TYPE_MAP[n.Metadata_Type__c];
            if (!typeMap[t]) typeMap[t] = new Set();
            typeMap[t].add(n.Metadata_Name__c);
        });

        const typeBlocks = Object.entries(typeMap).map(([type, names]) => {
            const members = [...names].sort().map(n => `        <members>${n}</members>`).join('\n');
            return `    <types>\n${members}\n        <name>${type}</name>\n    </types>`;
        }).join('\n');

        return `<?xml version="1.0" encoding="UTF-8"?>\n<Package xmlns="http://soap.sforce.com/2006/04/metadata">\n${typeBlocks}\n    <version>66.0</version>\n</Package>`;
    }

    _filename(ext, suffix) {
        return `${buildDefaultFilename(this.targetApiName, suffix || '')}.${ext}`;
    }

    _download(filename, mimeType, content) {
        const blob = new Blob([content], { type: mimeType });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        a.style.display = 'none';
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }

    _showError(message) {
        this.dispatchEvent(new CustomEvent('showerror', {
            detail: { message }, bubbles: true, composed: true
        }));
    }
}
```
